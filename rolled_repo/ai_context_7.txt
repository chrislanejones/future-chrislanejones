console.error("Failed to send message:", error);
setSubmitStatus("error");
} finally {
setIsSubmitting(false);
}
};

e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
) => {
const { name, value } = e.target;

setErrors((prev) => ({ ...prev, [name]: "" }));
}
};
