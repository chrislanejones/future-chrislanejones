| "yellow"
| "green"
| "pink"
| "purple"
| "cyan"
| "orange",
links: allLinks
.filter((link) => link.category === cat.category)
.sort((a, b) => a.order - b.order)
.map<BrowserLinkItem>((link) => ({
href: link.href,
label: link.label,
domain: link.domain,
favicon: link.favicon,
})),
}));

}
return (
<>
<div className="grid grid-cols-1 md:grid-cols-4 lg:grid-cols-6 gap-6 mt-10">

<BrowserCard
